using namespace std;

int main()
{
    int a, b, c; // Instantiating variables a, b and c

    cout << "Enter two integers to add\n";  
    cin >> a >> b;  // Taking user input for a and b

    c = a + b; // Adding a and b and storing the value in variable c
    cout <<"Sum of the numbers: " << c << endl; // Print the sum i.e. value of c   

    return 0;
}
